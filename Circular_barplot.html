plot_data = cbind(go_terms$Description, go_terms$log_pvalue_OL, go_terms$log_pvalue_BM)
plot_data = as.data.frame(plot_data)
colnames(plot_data) = c("GO_term","log.pval.OL","log.pval.BM")

plot_data$term = as.factor(plot_data$GO_term)
plot_data$log.pval.OL = as.numeric(plot_data$log.pval.OL)
plot_data$log.pval.BM = as.numeric(plot_data$log.pval.BM)


plot.data.sub = plot_data[1:15,]
# Transform data for ggplot
plot_data_long <- plot.data.sub %>%
  pivot_longer(cols = c(log.pval.OL, log.pval.BM), 
               names_to = "Sample", 
               values_to = "log.pval")

# Add an angle column for positioning
plot_data_long <- plot_data_long %>%
  arrange(GO_term, Sample) %>%
  mutate(
    angle = 90 - 360 * (1:n() - 0.5) / (n() / 2),  # Calculate angles for each bar
    hjust = ifelse(angle < -90, 1, 0),  # Adjust horizontal justification
    angle_label = ifelse(angle < -90, angle + 180, angle)  # Adjust angle for text
  )

p <- ggplot(plot_data_long, aes(x = GO_term, y = log.pval, fill = Sample)) +       
  geom_bar(stat = "identity", position = "dodge", alpha = 0.7) +  # Adjust position and transparency
  ylim(-4, 5) + 
  theme_minimal() +
  theme(
    axis.text = element_blank(),
    axis.title = element_blank(),
    panel.grid = element_blank(),
    plot.margin = unit(rep(-2, 4), "cm")
  ) +
  coord_polar(start = 0) +
  geom_text(aes(x = GO_term, y = log.pval + 0.5, label = GO_term, angle = angle_label, hjust = hjust),
            size = 1.5, color = "black") +  # Add labels
  scale_fill_manual(values = c("log.pval.OL" = "#CD5C5C", "log.pval.BM" = "#4783B5")) +  # Customize colors
  labs(title = "Circular Barplot of GO Term Significance",
       subtitle = "Comparing -log10(p-values) between OL and BM samples",
       fill = "Sample Origin")

print(p)
